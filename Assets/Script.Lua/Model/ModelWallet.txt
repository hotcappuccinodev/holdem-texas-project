MoneyType = {
    CNY = 0, -- 人民币
    USD = 1, -- 美元
    EUR = 2, -- 欧元
    NGN = 3, -- 尼日利亚，奈拉
    GHC = 4, -- 加纳，塞地

    BitCNY = 100, -- 虚拟币，BitCNY
}

WalletResult = {
    Success = 0,
    False = 1,
    Timeout = 2,
}

---------------------------------------
-- 充值请求
WalletRechargeRequest = {}

function WalletRechargeRequest:new(o)
    o = o or {}
    setmetatable(o, self)
    self.__index = self
    o.MoneyType = MoneyType.CNY
    o.Amount = 0
    o.Channel = nil
    return o
end

function WalletRechargeRequest:getData4Pack()
    local t = {}
    table.insert(t, self.MoneyType)
    table.insert(t, self.Amount)
    table.insert(t, self.Channel)

    return t
end

function WalletRechargeRequest:setData(data)
    self.MoneyType = data[1]
    self.Amount = data[2]
    self.Channel = data[3]
end

---------------------------------------
-- 充值通知
WalletRechargeNotify = {}

function WalletRechargeNotify:new(o)
    o = o or {}
    setmetatable(o, self)
    self.__index = self
    o.Result = WalletResult.Success
    o.Id = nil
    o.MoneyType = nil
    o.Address = nil
    o.Memo = nil
    o.Volume = nil
    o.Fee = nil
    return o
end

function WalletRechargeNotify:setData(data)
    self.Result = data[1]
    self.Id = data[2]
    self.MoneyType = data[3]
    self.Address = data[4]
    self.Memo = data[5]
    self.Volume = data[6]
    self.Fee = data[7]
end
